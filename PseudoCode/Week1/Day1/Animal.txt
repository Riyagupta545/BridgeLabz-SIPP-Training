CLASS Pet
    WORD name
    INTEGER age
    ATTRIBUTE breed

    CONSTRUCTOR Pet(name, age, breed)
        SET this.name = name
        SET this.age = age
        SET this.breed = breed

    METHOD displayInfo() 
        PRINT "Name: " + name
        PRINT "Age: " + age
        PRINT "Breed: " + breed


CLASS Dog EXTENDS Pet
    ATTRIBUTE isTrained

    CONSTRUCTOR Dog(name, age, breed, isTrained)
        CALL super(name, age, breed)
        SET this.isTrained = isTrained

    METHOD displayInfo()
        CALL super.displayInfo()
        PRINT "Trained: " + isTrained


    METHOD add
            INTEGER a
            INTEGER b

            RETUEN a+b
